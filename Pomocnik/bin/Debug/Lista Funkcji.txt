### Lista funkcji:
Obsluga_czasu:
	- bool Uspij(...)									// Uśpienie programu na x sekund
	- string Podaj_date_i_czas(...)						// Zwraca obecny czas, tryby: 0-YYYYMMDDGGmmSS, 1-YYYYMMDD, 2-GGmmSS

Obsluga_sieci:
	- string Wyszukaj_na_stronie(...)					// Wyszukaj linie pod danym adresem i usuń podany tekst na początku i na końcu
	- bool Sprawdz_polaczenie_z_internetem()			// Sprawdź czy można połączyć z internetem

Obsluga_plikow:
	- string[,] Wczytaj_plik_tekstowy(...)										// Wczytaj plik (w postaci "cały plik" ==> "oddzielne linie"), tablica 2d
	- string[] Sprawdz_komentarze(...)					// Usuń komentarze (zaczyna się od "komentarz")
	- string[,] Podziel_na_kolumny(...)					// Podziel tablicę 1d na 2d przez ","

Obsluga_tekstu:
	- string Znajdz_i_zamien(...)						// Znajdz i zamień szukany tekst na inny
